# I'm curious about the difference in realfeel and actual temp, and correlation
# with windspeed etc.

colnames(b)[length(b)] <- "tempdiff"

bc <- cor(b |> select(-date)) # can't have strings
# https://cran.r-project.org/web/packages/corrplot/vignettes/corrplot-intro.html
# tl.pos = 'd', cl.pos = 'n' #move labels to diag

#realfeel, season, weather, windspeed

mod1 <- lm(data = b, rentals~realfeel)
